```bash
# 安裝 aws cli 
curl "https://awscli.amazonaws.com/awscli-exe-linux-x86_64.zip" -o "awscliv2.zip"
unzip awscliv2.zip
sudo ./aws/install

# 設定，完成後會出現 ~/.aws，內含 credentials
aws configure
```

C++ SDK

```
sudo apt-get update
sudo apt-get install cmake g++ libssl-dev libcurl4-openssl-dev zlib1g-dev

git clone --recurse-submodules https://github.com/aws/aws-sdk-cpp.git
cd aws-sdk-cpp
mkdir build && cd build
cmake .. -DBUILD_ONLY="s3" -DCMAKE_BUILD_TYPE=Release
make
sudo make install
```
