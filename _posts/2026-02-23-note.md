### False sharing 相關程式碼 待確認 有點怪

```cpp

#include <iostream>
#include <thread>
#include <atomic>
#include <chrono>

// 查看系統的 cache line size
void print_cache_info() {
    std::cout << "Cache line size: " 
              << std::hardware_destructive_interference_size 
              << " bytes\n";  // C++17, 通常是 64 bytes
    
    // 或使用 Linux sysfs
    system("cat /sys/devices/system/cpu/cpu0/cache/index0/coherency_line_size");
}

// ❌ BAD: False Sharing 範例
struct BadCounters {
    std::atomic<uint64_t> counter1;  // offset 0
    std::atomic<uint64_t> counter2;  // offset 8
    // counter1 和 counter2 在同一個 cache line (64 bytes)
    // 不同 CPU 修改時會造成 cache line bouncing
};

void test_false_sharing() {
    BadCounters counters{};
    
    auto worker1 = [&]() {
        for (int i = 0; i < 100000000; ++i) {
            counters.counter1.fetch_add(1, std::memory_order_relaxed);
        }
    };
    
    auto worker2 = [&]() {
        for (int i = 0; i < 100000000; ++i) {
            counters.counter2.fetch_add(1, std::memory_order_relaxed);
        }
    };
    
    auto start = std::chrono::high_resolution_clock::now();
    
    std::thread t1(worker1);
    std::thread t2(worker2);
    t1.join();
    t2.join();
    
    auto end = std::chrono::high_resolution_clock::now();
    auto duration = std::chrono::duration_cast<std::chrono::milliseconds>(end - start);
    
    std::cout << "False sharing test: " << duration.count() << " ms\n";
}

// ✅ GOOD: Cache Line Alignment
struct alignas(64) GoodCounters {
    std::atomic<uint64_t> counter1;
    char padding1[64 - sizeof(std::atomic<uint64_t>)];
    
    std::atomic<uint64_t> counter2;
    char padding2[64 - sizeof(std::atomic<uint64_t>)];
    // counter1 和 counter2 在不同的 cache lines
};

void test_no_false_sharing() {
    GoodCounters counters{};
    
    auto worker1 = [&]() {
        for (int i = 0; i < 100000000; ++i) {
            counters.counter1.fetch_add(1, std::memory_order_relaxed);
        }
    };
    
    auto worker2 = [&]() {
        for (int i = 0; i < 100000000; ++i) {
            counters.counter2.fetch_add(1, std::memory_order_relaxed);
        }
    };
    
    auto start = std::chrono::high_resolution_clock::now();
    
    std::thread t1(worker1);
    std::thread t2(worker2);
    t1.join();
    t2.join();
    
    auto end = std::chrono::high_resolution_clock::now();
    auto duration = std::chrono::duration_cast<std::chrono::milliseconds>(end - start);
    
    std::cout << "No false sharing test: " << duration.count() << " ms\n";
}

int main() {
    print_cache_info();
    test_false_sharing();        // 預期: ~800-1200ms
    test_no_false_sharing();     // 預期: ~200-400ms (快 2-4x!)
    return 42;
}
```


### CPU cycles 計數 & 計時相關

```cpp


#include <iostream>
#include <stdint.h>
// #include <x86intrin.h>   // __rdtsc() 也在這裡，但我們這裡用自己的版本
#include <unistd.h>      // usleep()
#include <chrono>

// 手動實作 rdtsc
inline uint64_t rdtsc() {
    uint32_t lo, hi;
    __asm__ __volatile__("rdtsc" : "=a"(lo), "=d"(hi));
    return ((uint64_t)hi << 32) | lo;
}

int main() {
    std::cout << "Starting RDTSC test...\n";

    // 預估 CPU 頻率（GHz）——建議手動填寫你的 CPU 實際值
    double cpu_freq_ghz = 3.5; // 例如 3.5 GHz

    uint64_t start = rdtsc();

    // 要測的工作區段
    volatile uint64_t sum = 0;
    for (uint64_t i = 0; i < 1000000; ++i) {
        sum += i;
    }

    uint64_t end = rdtsc();

    uint64_t cycles = end - start;
    double seconds = cycles / (cpu_freq_ghz * 1e9);

    std::cout << "Sum: " << sum << "\n";
    std::cout << "CPU cycles: " << cycles << "\n";
    std::cout << "Approx time: " << seconds << " sec" << std::endl;

    return 0;
}
```


### Read-Write Lock

```cpp

#include <pthread.h>
#include <stdio.h>
#include <unistd.h>

pthread_rwlock_t rwlock;
int shared_data = 0;

void* reader(void* arg) {
    int id = (int)(long)arg;
    for (int i = 0; i < 3; ++i) {
        pthread_rwlock_rdlock(&rwlock); // 取得讀鎖
        printf("Reader %d: read shared_data = %d\n", id, shared_data);
        pthread_rwlock_unlock(&rwlock); // 解鎖
        sleep(1);
    }
    return NULL;
}

void* writer(void* arg) {
    int id = (int)(long)arg;
    for (int i = 0; i < 3; ++i) {
        pthread_rwlock_wrlock(&rwlock); // 取得寫鎖
        shared_data++;
        printf("Writer %d: updated shared_data = %d\n", id, shared_data);
        pthread_rwlock_unlock(&rwlock); // 解鎖
        sleep(2);
    }
    return NULL;
}

int main() {
    pthread_t r1, r2, w1;
    pthread_rwlock_init(&rwlock, NULL);

    pthread_create(&r1, NULL, reader, (void*)1);
    pthread_create(&r2, NULL, reader, (void*)2);
    pthread_create(&w1, NULL, writer, (void*)1);

    pthread_join(r1, NULL);
    pthread_join(r2, NULL);
    pthread_join(w1, NULL);

    pthread_rwlock_destroy(&rwlock);
    return 0;
}
```


### Pthread Barrier

```cpp
#include <pthread.h>
#include <iostream>
#include <vector>
#include <unistd.h>

pthread_barrier_t barrier;

void* worker(void* arg) {
    int id = *(int*)arg;

    for (int round = 1; round <= 3; ++round) 
    {
        pthread_barrier_wait(&barrier);

        usleep(500*id + 1);
        std::cout << "Thread " << id << " round " << round << "\n";

        pthread_barrier_wait(&barrier);  // 等所有人完成 → 再進下一輪
    }

    return nullptr;
}

int main() {
    int n = 3;
    pthread_barrier_init(&barrier, nullptr, n);

    std::vector<pthread_t> threads(n);
    std::vector<int> ids(n);
    for (int i = 0; i < n; i++) {
        ids[i] = i;
        pthread_create(&threads[i], nullptr, worker, &ids[i]);
    }
    for (int i = 0; i < n; i++) {
        pthread_join(threads[i], nullptr);
    }

    pthread_barrier_destroy(&barrier);
}
```
